Widget getVMSelector(int index) {
  if (index >= vmGlobalList.length) return Container();
  return GestureDetector(
    onTap: () {
      setEnvironment(vmGlobalList[index].code);

      if (selectedVMIndex == index) {
        selectedVMIndex = -1;
      } else {
        selectedVMIndex = index;
      }

      if (selectedVMIndex == 0) {
        context.read<StepperCubit>().changeSteps(initialSteps);
      } else {
        context.read<StepperCubit>().changeSteps(cloudSteps);
      }
      setState(() {});
    },
    child: Container(
      height: 144 * hs,
      width: 188 * ws,
      padding: EdgeInsets.only(
        left: 21 * ws,
        right: 27 * ws,
      ),
      decoration: ShapeDecoration(
        shape: RoundedRectangleBorder(
          side: selectedVMIndex == index ? selectedSide : nonSelectedSide,
          borderRadius: BorderRadius.circular(8),
        ),
      ),
      child: Column(
        mainAxisSize: MainAxisSize.min,
        mainAxisAlignment: MainAxisAlignment.start,
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          SizedBox(height: 21 * hs),
          Align(
            alignment: Alignment.centerRight,
            child: selectedVMIndex == index
                ? getSelectedRadio()
                : getNonSelectedRadio(),
          ),
          SizedBox(height: 4 * hs),
          Container(
            width: vmGlobalList[index].w,
            height: vmGlobalList[index].h,
            decoration: BoxDecoration(
              color: Colors.white,
              image: DecorationImage(
                image: AssetImage(vmGlobalList[index].path),
                fit: BoxFit.fill,
              ),
            ),
          ),
          SizedBox(height: vmGlobalList[index].spaceWidth),
          Text(
            vmGlobalList[index].name,
            textAlign: TextAlign.center,
            style: TextStyle(
              color: const Color(0xFF636366),
              fontSize: 12 * ws,
              fontFamily: 'Heebo',
              fontWeight: FontWeight.w500,
            ),
          ),
        ],
      ),
    ),
  );
}
