void upload(Uint8List fileBytes) async {
  emit(LicenseFileUploading());

  // Mocking Upload percentage
  increseUploadPercentage();

  final response = await sl<Repository>().uploadLicense(fileBytes);
  response.fold(
    (l) => emit(LicenseUploadError(failure: l)),
    (r) async {
      // Making the upload percent as 100 before chaning the state to uploaded
      emit(LicenseFileUploading(uploadPercentage: 100));
      await Future.delayed(const Duration(milliseconds: 120));

      emit(LicenseFileUploaded());
    },
  );
}
