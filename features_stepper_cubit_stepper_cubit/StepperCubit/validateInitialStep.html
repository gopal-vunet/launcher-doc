static String validateInitialStep(
  String providedInitialStep,
  SavedConfiguration savedConfiguration,
) {
  // Checking if the first step is filled out
  final selectedEnv = savedConfiguration.savedConfig['vm_selection'];
  if (selectedEnv == null) {
    // First step is empty, should redirect to first step
    return '/vm_selection';
  }

  // First step is filled, checking the next steps
  late List<String> steps;
  if (selectedEnv == 'vm') {
    steps = nonCloudSteps;
  } else {
    steps = cloudSteps;
  }
  // Here index starts at 1 as the first step in already validated above
  for (var i = 1; i < steps.length; i++) {
    // If the [providedInitialStep] is same as currently being validated step
    // it can be allowed be routed
    if (providedInitialStep == steps[i]) return providedInitialStep;
    final savedConfigForCurrentRoute = savedConfiguration
        .savedConfig[StepperCubit.stripFirstCharacterIfSlash(steps[i])];
    if (savedConfigForCurrentRoute == null) {
      // The current step being iterated is not filled out, so
      // redirection ha to be made to this route
      return steps[i];
    }
  }
  return '/vm_selection';
}
