void addReviewStage(Map<String, dynamic> data) {
  reviewStages = [];
  try {
    // Add license Information
    reviewStages.add(
      ReviewStage(
        stageName: "License Information",
        stageContent: Map<String, String>.from(data["license_info"]),
      ),
    );

    // Add Environment type
    reviewStages.add(ReviewStage(
        stageName: "Environment Type",
        stageContent: Map<String, String>.from(data["env_info"])));

    // Add VM Configuration
    reviewStages.add(ReviewStage(
        stageName: "VM Configuration",
        stageContent: Map<String, dynamic>.from(data["vm_configuration"])));

    // Add ServiceVM Mapping
    reviewStages.add(ReviewStage(
        stageName: "Service VM Allocation",
        stageContent: Map<String, dynamic>.from(data["service_mapper"])));

    // Add port information
    reviewStages.add(ReviewStage(
        stageName: "Port Information",
        stageContent: Map<String, int>.from(data["port_configuration"])));
  } catch (e) {
    logger.e("Exception - $e");
  }
}
