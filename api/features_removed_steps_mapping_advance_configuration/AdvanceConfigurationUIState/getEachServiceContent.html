Widget getEachServiceContent(ServicePlaceHolder service) {
  return Row(
    crossAxisAlignment: CrossAxisAlignment.start,
    children: [
      Container(
        width: 150 * ws,
        padding: EdgeInsets.only(top: 20 * hs),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(
              service.serviceName,
              style: TextStyle(
                color: Colors.black,
                fontSize: 18 * ws,
                fontFamily: 'Heebo',
                fontWeight: FontWeight.w800,
                letterSpacing: -0.40,
              ),
            ),
          ],
        ),
      ),
      SizedBox(width: 66 * ws),
      Expanded(
        child: Container(
          padding: EdgeInsets.only(top: 20 * hs),
          constraints: BoxConstraints(
            maxWidth: 1000 * ws,
          ),
          child: Wrap(
            direction: Axis.horizontal,
            alignment: WrapAlignment.start,
            children: [
              ...List.generate(service.vmList.length, (index) {
                return Padding(
                  padding: EdgeInsets.only(bottom: 16 * hs, right: 38 * ws),
                  child: getSingleVM(
                    service.vmList[index].hostname,
                    service.vmList[index].ipAddress,
                  ),
                );
              }),
            ],
          ),
        ),
      ),
      IconButton(
        onPressed: () {
          showAlert(service);
        },
        icon: Icon(
          Icons.edit_outlined,
          size: 31 * ws,
          color: Colors.black,
        ),
      ),
    ],
  );
}
