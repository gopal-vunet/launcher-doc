Widget getContinueButton(BuildContext context, InstallEvent event) {
  bool isValid = false;

  String buttonName = "Cancel Installation";
  if (event.hasAnyEventFailed()) {
    buttonName = "Retry";
  } else if (event.isAllEventSucceededGlobal()) {
    isValid = true;
    buttonName = "Go to vuSmartMaps";
  }
  return ElevatedButton(
    onPressed: () {
      // if user tries to retry the failed stage, then enable the alertWindow
      if (buttonName == "Retry") {
        // showAlertWindow = true;

        // retryInstallation();
        context.read<InstallationCubit>().startInstallation();
      } else if (buttonName == "Cancel Installation") {
        final installationState = context.read<InstallationCubit>().state;
        if (installationState is Installing) {
          showCancelConfirmAlert(installationState);
        }
      } else if (buttonName == "Go to vuSmartMaps") {
        moveToVuSmartMap();
      }

      // Navigator.pushNamed(context, "/installation_progress");
    },
    style: ElevatedButton.styleFrom(
      backgroundColor: !isValid
          ? const Color(0xFFEB0013)
          : const Color.fromRGBO(17, 51, 154, 1),
      shape: RoundedRectangleBorder(
        borderRadius: BorderRadius.circular(8),
      ),
      minimumSize: const Size(333, 58),
      maximumSize: const Size(333, 58),
    ),
    child: Row(
      mainAxisAlignment: MainAxisAlignment.spaceEvenly,
      children: [
        Text(
          buttonName,
          style: const TextStyle(
            color: Colors.white,
            fontSize: 18,
            fontFamily: 'Heebo',
            fontWeight: FontWeight.w500,
            letterSpacing: -0.40,
          ),
        ),
      ],
    ),
  );
}
