Widget getSuggestionWindow() {
  List<VMInformation> filteredList = getFilteredSuggestion();
  return ConstrainedBox(
    constraints: const BoxConstraints(
      maxHeight: 150,
      minHeight: 0,
    ),
    child: Container(
      width: 328,
      decoration: BoxDecoration(
        border: Border.all(
          color: const Color(0xFFD8DEE4),
          width: 1.0, // Border width
        ),
        borderRadius: BorderRadius.circular(4.0), // Border radius
        color: Colors.white, // Background color
      ),
      child: ListView.builder(
        shrinkWrap: true,
        itemCount: filteredList.length,
        itemBuilder: (context, index) {
          return
              // NOTE:
              // Ontap doesn't work. Because, we add this widget on top of another, for some reason this doesn't
              // outer widget doesn't detect the click. Ref : https://github.com/flutter/flutter/issues/55449
              // hence using onPanDown as the workaround.
              GestureDetector(
            onPanDown: (_) {
              copyServiceInfo.vmList[getVMIndex(filteredList[index].hostname)]
                  .isVisible = true;
              setState(() {});
            },
            child: ListTile(
              title: Text(
                "${copyServiceInfo.vmList[index].hostname}       ${copyServiceInfo.vmList[index].ipAddress}",
                style: const TextStyle(
                  color: Colors.black,
                  fontSize: 13,
                  fontFamily: 'Heebo',
                  fontWeight: FontWeight.w400,
                ),
              ),
            ),
          );
        },
      ),
    ),
  );
}
