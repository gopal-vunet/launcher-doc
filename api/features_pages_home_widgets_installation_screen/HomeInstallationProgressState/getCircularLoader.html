Widget getCircularLoader() {
  return Stack(
    alignment: Alignment.center,
    children: [
      SizedBox(
        width: 102.0,
        height: 102.0,
        child: CircularProgressIndicator(
          value: widget.event?.getTotalPercentage().toDouble() ?? 0 / 100.0,
          strokeWidth: 12.0, // Adjust the strokeWidth as needed
          valueColor: AlwaysStoppedAnimation<Color>(
            widget.event?.hasAnyEventFailed() ?? false
                ? Colors.red
                : const Color.fromRGBO(0, 164, 58, 1),
          ),

          backgroundColor: const Color.fromRGBO(237, 237, 236, 1),
        ),
      ),
      Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          Text(
            '${widget.event?.getTotalPercentage()}%',
            style: const TextStyle(
              color: Colors.black,
              fontSize: 18,
              fontFamily: 'Heebo',
              fontWeight: FontWeight.w500,
            ),
          ),
          const SizedBox(height: 4),
          const Text(
            'Complete',
            style: TextStyle(
              color: Colors.black,
              fontSize: 12,
              fontFamily: 'Heebo',
              fontWeight: FontWeight.w400,
            ),
          ),
        ],
      ),
    ],
  );
}
