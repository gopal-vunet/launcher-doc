void updateEvent(List<Map<String, dynamic>> eventList) {
  if (!canEventProcessed) {
    return;
  }
  installationEvents.clear();

  // add events to the list
  for (var event in eventList) {
    installationEvents.add(EventDetails(
      stageName: event["stage_name"].toString(),
      isStarted: event["is_started"] as bool,
      isSucceeded: event["is_succeeded"] as bool,
      isCompleted: event["is_completed"] as bool,
      retryCount: event["retry_count"] as int,
      description: event["description"].toString(),
      index: event["index"] as int,
    ));
  }
  // Come up with better approach when you get some time.
  installationEvents.sort((a, b) => a.index.compareTo(b.index));
  // refresh the UI
  reload();
}
