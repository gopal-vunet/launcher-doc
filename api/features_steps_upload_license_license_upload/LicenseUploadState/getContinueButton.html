Widget getContinueButton(BuildContext context, bool isValid) {
  return ElevatedButton(
    onPressed: () {
      if (isValid) context.read<StepperCubit>().moveToNextStep(context);
    },
    style: ElevatedButton.styleFrom(
      backgroundColor: isValid
          ? const Color.fromRGBO(17, 51, 154, 1)
          : const Color.fromRGBO(225, 225, 225, 1),
      shape: RoundedRectangleBorder(
        borderRadius: BorderRadius.circular(8),
      ),
      minimumSize: Size(333 * ws, 58 * hs),
    ),
    child: Text(
      'Continue',
      style: TextStyle(
        color: Colors.white,
        fontSize: 18 * ws,
        fontFamily: 'Heebo',
        fontWeight: FontWeight.w500,
        height: 1.67 * hs,
        letterSpacing: -0.40,
      ),
    ),
  );
}
