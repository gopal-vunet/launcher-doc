factory Message.fromMap(Map<String, dynamic> map) {
  return Message(
    stageName: map['stage_name'] ?? '',
    eventName: map['event_name'] ?? '',
    description: map['description'] ?? '',
    userId: map['user_id'] ?? '',
    envName: map['env_name'] ?? '',
    retryCount: map['retry_count']?.toInt() ?? 0,
    orderNum: map['order_num']?.toInt() ?? 0,
    isStarted: map['is_started'] ?? false,
    isCompleted: map['is_completed'] ?? false,
    isSucceeded: map['is_succeeded'] ?? false,
  );
}
