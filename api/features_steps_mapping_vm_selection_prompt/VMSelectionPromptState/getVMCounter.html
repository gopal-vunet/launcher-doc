Widget getVMCounter(VMInformation vmInfo) {
  return Row(
    mainAxisAlignment: MainAxisAlignment.spaceBetween,
    children: [
      IconButton(
        onPressed: () {
          vmInfo.count--;
          vmInfo.count = max(0, vmInfo.count);
          setState(() {});
        },
        icon: const Icon(
          Icons.remove_circle_outlined,
          size: 33,
          color: Color.fromRGBO(33, 90, 255, 1),
        ),
      ),
      const SizedBox(width: 9.5),
      Align(
        alignment: const Alignment(0, 0.5),
        child: Text(
          '${vmInfo.count}',
          style: const TextStyle(
            color: Color(0xFF313043),
            fontSize: 22,
            fontFamily: 'Poppins',
            fontWeight: FontWeight.w500,
            letterSpacing: 0.22,
          ),
        ),
      ),
      IconButton(
        onPressed: () {
          vmInfo.count++;
          setState(() {});
        },
        icon: const Icon(
          Icons.add_circle_outlined,
          size: 33,
          color: Color.fromRGBO(33, 90, 255, 1),
        ),
      ),
      const SizedBox(width: 13),
      getDeleteIcon(vmInfo),
    ],
  );
}
