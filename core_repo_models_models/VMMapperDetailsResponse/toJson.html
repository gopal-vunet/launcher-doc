Map<String, dynamic> toJson() {
  var serviceVmMapperToJson = serviceVmMapper.map(
    (key, value) => MapEntry(
      key,
      value.map((item) => item.toJson()).toList(),
    ),
  );

  return {
    'service_count': serviceCount,
    'vm_list': vmList,
    'service_vm_mapper': serviceVmMapperToJson,
  };
}
