Widget getSecretHolder(int index) {
  return GestureDetector(
    onTap: () {
      selectedSecret = index;
      setState(() {});
    },
    child: Container(
      height: 88 * hs,
      width: 380 * ws,
      padding: EdgeInsets.symmetric(horizontal: 24 * ws, vertical: 24 * hs),
      decoration: ShapeDecoration(
        color: Colors.white,
        shape: RoundedRectangleBorder(
          side: selectedSecret == index ? selectedSide : nonSelectedSide,
          borderRadius: BorderRadius.circular(8),
        ),
      ),
      child: Row(
        mainAxisSize: MainAxisSize.min,
        mainAxisAlignment: MainAxisAlignment.spaceBetween,
        crossAxisAlignment: CrossAxisAlignment.center,
        children: [
          Column(
            mainAxisSize: MainAxisSize.min,
            mainAxisAlignment: MainAxisAlignment.start,
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Text(
                secretType[index].item1,
                textAlign: TextAlign.center,
                style: TextStyle(
                  color: selectedSecret == index
                      ? Colors.black
                      : const Color(0xFF636366),
                  fontSize: 14 * ws,
                  fontFamily: 'Heebo',
                  fontWeight: FontWeight.w900,
                ),
              ),
              SizedBox(
                height: 4 * hs,
              ),
              Text(
                secretType[index].item2,
                textAlign: TextAlign.center,
                style: TextStyle(
                  color: const Color(0xFF636366),
                  fontSize: 10 * ws,
                  fontFamily: 'Heebo',
                  fontWeight: FontWeight.w400,
                ),
              ),
            ],
          ),
          //const SizedBox(width: 377),
          selectedSecret == index
              ? getSelectedRadio()
              : getNonSelectedRadio(),
        ],
      ),
    ),
  );
}
